        function switchNivel1(idContainer, btn) {
            document.querySelectorAll('.n1-container').forEach(el => el.classList.add('hidden'));
            document.getElementById(idContainer).classList.remove('hidden');
            
            document.querySelectorAll('.btn-n1').forEach(b => b.classList.remove('bg-white/20', 'font-bold'));

        function switchNivel2(idContainer, containerClass, btn) {
            document.querySelectorAll('.' + containerClass).forEach(el => el.classList.add('hidden'));

            const pai = btn.parentElement;

        }
